int ensure_capacity(char **buf, size_t *cap, size_t need) {
    if (need <= *cap) {
        return 1;
    }

    while (*cap < need) {
        *cap *= 2;
    }

    char *tmp = realloc(*buf, *cap);
    if (tmp == NULL) {
        return 0;
    }

    *buf = tmp;
    return 1;
}
